2. ARCHITECTURE MEMOIRE RAM vs ROM

Philosophie: Memoire Vivante vs Memoire Permanente

La memoire d'un systeme IA autonome fonctionne comme un cerveau humain: la RAM porte

le contexte immediat de la session, la ROM stocke la sagesse accumulee qui survit aux

redemarrages. Sans cette separation, chaque session Claude repart de zero -- inacceptable

pour une mission a 63M‚Ç™.

RAM -- Memoire Volatile de Session


RAM LAYER (session active uniquement)
+-- Redis Enterprise (in-memory, localhost:6379)
|   +-- session:{session_id}          -&gt; contexte JSON courant (TTL 4h)
|   +-- rate_limit:{agent}:{service}  -&gt; compteur (TTL 60s)
|   +-- lock:{resource}               -&gt; mutex distribue (TTL 30s)
|   +-- queue:{priority}              -&gt; OPUS_QUEUE items en cours
|   +-- token:{service}               -&gt; tokens auth hot-loaded (TTL 24h)
|
+-- SESSION_LIVE.md
|   Chemin: ~/Desktop/DREAMNOVA_MISSION_CONTROL/SESSION_LIVE.md
|   Format: ### HH:MM -- [action] + bullets concis
|   Flush: toutes les 10 actions significatives (OBLIGATOIRE)
|
+-- OPUS_QUEUE.md (taches en cours)
    Chemin: ~/scripts/opus-loop/OPUS_QUEUE.md
    Format: TODO: [tache] -&gt; DONE: [tache] [YYYY-MM-DD HH:MM]

Regle RAM critique: Avant toute operation longue (>5 min), figer l'etat dans

SESSION_LIVE.md. Si crash ou coupure courant: ce fichier est le seul moyen de reprendre.

ROM -- Memoire Permanente Inter-Sessions


ROM LAYER (persiste entre sessions, survit aux redemarrages)
+-- MEMORY.md
|   Chemin: ~/.claude/projects/-Users-codenolimits-dreamai-nanach/memory/MEMORY.md

+-- NUCLEUS.md

|   Taille: &lt; 2400 chars (compact, charge EN PREMIER chaque session)

+-- Milvus Vector DB (fichier local, pas de serveur separe)
|   URI: ./milvus_local.db
|   Index: HNSW (Hierarchical Navigable Small World)

Code: Gestionnaire Memoire Hybride


# ~/scripts/memory_manager.py
import redis
import asyncio
import hashlib
import json
from datetime import datetime
from pathlib import Path
from pymilvus import MilvusClient
from openai import AsyncOpenAI

REDIS_URL = "redis://localhost:6379"
MILVUS_URI = str(Path.home() / "milvus_local.db")
SESSION_LIVE_PATH = (
    Path.home() / "Desktop/DREAMNOVA_MISSION_CONTROL/SESSION_LIVE.md"
)


class DreamNovaMemory:
    """Gestionnaire memoire hybride RAM (Redis) + ROM (Milvus + fichiers)."""

    def __init__(self):
        self.redis = redis.from_url(REDIS_URL, decode_responses=True)
        self.milvus = MilvusClient(MILVUS_URI)
        self.openai = AsyncOpenAI()
        self._ensure_collections()

    def _ensure_collections(self):
        collections = [
            "emails_clients", "conversations_wa",
            "knowledge_torah", "code_snippets",
        ]
        for col in collections:
            if not self.milvus.has_collection(col):
                self.milvus.create_collection(
                    collection_name=col,
                    dimension=1536,
                    metric_type="COSINE",
                    index_type="HNSW",
                    params={"M": 16, "efConstruction": 200},
                )

    async def embed(self, text: str) -&gt; list[float]:
        resp = await self.openai.embeddings.create(
            model="text-embedding-3-small", input=text[:8192]
        )
        return resp.data[0].embedding

    async def store_rom(self, collection: str, text: str, metadata: dict):
        embedding = await self.embed(text)
        doc_id = int(hashlib.md5(text.encode()).hexdigest()[:8], 16)
        self.milvus.insert(
            collection,
            [{"id": doc_id, "vector": embedding, "text": text, **metadata}],
        )

    async def recall_rom(
        self, collection: str, query: str, top_k: int = 5
    ) -&gt; list[dict]:
        embedding = await self.embed(query)
        return self.milvus.search(
            collection_name=collection,
            data=[embedding],
            limit=top_k,
            output_fields=["text", "source", "timestamp"],
        )[0]

    def store_ram(self, key: str, value: str, ttl_seconds: int = 14400):
        self.redis.setex(key, ttl_seconds, value)

    def recall_ram(self, key: str) -&gt; str | None:
        return self.redis.get(key)

    def flush_session_to_live(self, action: str, bullets: list[str]):
        timestamp = datetime.now().strftime("%H:%M")
        lines = [f"\n### {timestamp} -- {action}"]
        lines += [f"- {b}" for b in bullets]
        lines.append("")
        with open(SESSION_LIVE_PATH, "a", encoding="utf-8") as f:
            f.write("\n".join(lines))

    def freeze_state(self, label: str, data: dict):
        self.redis.setex(
            f"freeze:{label}", 3600,
            json.dumps(data, ensure_ascii=False),
        )

    def restore_state(self, label: str) -&gt; dict | None:
        raw = self.redis.get(f"freeze:{label}")
        return json.loads(raw) if raw else None

3.2 Memubot -- Vision Proactive YOLOv10

Memubot observe l'ecran en permanence via YOLOv10 + OCR et anticipe les actions de David.

Capacites cles:

  • analyze_screen_buffer(): lecture pixels temps reel, detection UI elements
  • extract_2fa_sms(): scrape SMS apps, retourne code OTP automatiquement
  • prefill_form(url, data): remplit formulaires avant que David les ouvre
  • detect_routine(time_range): apprend les habitudes (9h = emails, 14h = code)
  • reduce_tokens(context): compresse contexte en supprimant elements hors-focus

# Usage type Memubot
from memubot import MemuBot

bot = MemuBot()

async def auto_2fa_workflow(service: str) -&gt; str:
    bot.start_screen_watch(interval_ms=500)
    code = await bot.extract_2fa_sms(
        source_app="Messages",
        service_pattern=service,
        timeout_seconds=60,
    )
    bot.stop_screen_watch()

async def create_stealth_browser(headless: bool = False):
    playwright = await async_playwright().start()
    browser = await playwright.chromium.launch(
        headless=headless,
        args=[
            "--disable-blink-features=AutomationControlled",
            "--disable-dev-shm-usage",
            "--no-sandbox",
        ],
    )
    context = await browser.new_context(
        locale="he-IL",
        timezone_id="Asia/Jerusalem",
        color_scheme="dark",
        extra_http_headers={"Accept-Language": "he-IL,fr;q=0.9,en;q=0.8"},
    )
    # Masquage WebGL/Canvas via page.add_init_script
    vendor = STEALTH_OVERRIDES["webgl_vendor"]
    renderer = STEALTH_OVERRIDES["webgl_renderer"]
    await context.add_init_script(f"""
        Object.defineProperty(navigator, 'webdriver', {{get: () =&gt; undefined}});
        const getParam = WebGLRenderingContext.prototype.getParameter;
        WebGLRenderingContext.prototype.getParameter = function(p) {{
            if (p === 37445) return '{vendor}';
            if (p === 37446) return '{renderer}';
            return getParam.call(this, p);
        }};
    """)
    return browser, context

async def human_move_to(page, selector: str):
    element = await page.query_selector(selector)
    box = await element.bounding_box()
    target_x = box["x"] + box["width"] / 2
    target_y = box["y"] + box["height"] / 2
    # Courbe de Bezier: 12 points avec variation aleatoire
    for i in range(12):
        t = i / 12
        x = 1280 + (target_x - 1280) * t + random.gauss(0, 3)
        y = 720 + (target_y - 720) * t + random.gauss(0, 3)
        await page.mouse.move(x, y)
        await asyncio.sleep(random.uniform(0.01, 0.03))
    await page.mouse.click(target_x, target_y)
